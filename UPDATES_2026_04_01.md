
```bash

# 2. Нажмите Download → CSV
# 3. Сохраните файл
```

### Вариант 2: SQL миграция (для разработчиков)

```sql
-- Скопировать данные из старой таблицы в новую
INSERT INTO applications (nickname, age, experience, rank, discord, about, created_at)
SELECT 
  pubg_nickname as nickname,
  CAST(age AS TEXT) as age,
  'N/A' as experience,
  rank,
  discord_telegram as discord,
  CONCAT(why_join_clan, ' ', what_can_provide) as about,
  created_at
FROM player_applications;

-- Удалить старую таблицу
DROP TABLE player_applications CASCADE;
```
